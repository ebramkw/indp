319 /* -------------------------- Clock Capture ------------------------- */
325 #define CAPTURE_NEXT_CLOCK_TICK(t_cap_h, t_cap_l) do {\
326  /* Enable capture mode for timers B6 and A2 (ACLK) */\
327  TBCCTL6 = CCIS0 | CM_POS | CAP | SCS; \
328  TACCTL2 = CCIS0 | CM_POS | CAP | SCS; \
329  /* Wait until both timers capture the next clock tick */\
330  while (!((TBCCTL6 & CCIFG) && (TACCTL2 & CCIFG))); \
331  /* Store the capture timer values */\
332  t_cap_h = TBCCR6; \
333  t_cap_l = TACCR2; \
334  /* Disable capture mode */\
335  TBCCTL6 = 0; \
336  TACCTL2 = 0; \
337 } while (0)
